while True:
    try:
        n = int(input("숫자를 입력하세요 (1-9): "))
        if 1 <= n <= 9:
            break
        else:
            print("1부터 9 사이의 숫자를 입력해주세요")
    except ValueError:
        print("유효한 숫자를 입력해주세요")

num = 1
a = 2

for i in range(1, n + 1):
    print(" " * a, end="") 
    for j in range(2 * i - 1):
        if num > n:
            break
        print(num, end="")
        num += 1
    print() 
    a -= 1
    if num > n:
        break
