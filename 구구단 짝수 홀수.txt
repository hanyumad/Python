# 2-12-1 while 반복문 구구단 5단 출력
# 홀수

a = 5
i = 1

while i <= 9:
  print(str(a) + 'X' + str(i) + ' = ' + str(i * a))
  i += 2
print('파이썬으로 구구단 5단을 계산할 수 있다!')

#짝수

a = 5
i = 1

while i <= 9:
  print(str(a) + 'X' + str(i + 1) + ' = ' + str((i + 1) * a))
  i += 2
print('파이썬으로 구구단 5단을 계산할 수 있다!')
