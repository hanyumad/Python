# 과제
# 2-12-1 while 반복문 구구단 출력

# 홀수

a = 2

while a <= 9:
    i = 1
    while i <= 9:
        print(str(a) + 'X' + str(i) + ' = ' + str(i*a))
        i += 2
    print()
    a += 1

print('파이썬으로 구구단 홀수를 계산할 수 있다!')
print()

# 짝수

a = 2

while a <= 9:
    i = 2
    while i <= 9:
        print(str(a) + 'X' + str(i) + ' = ' + str(i*a))
        i += 2
    print()
    a += 1

print('파이썬으로 구구단 짝수를 계산할 수 있다!')
print()
